//Insertion Sorting Algorithm
// Time complexity= O(n^2)
// Space complexity= O(1)

#include <stdio.h>
#define maxsz 10
void insertionsort(int a[], int n) 
{
    int i, j;
    int hold;                 // hold variable temporarily stores the value of current element being considered fro insertion.

    for (i = 1; i < n; i++) 
    {
        hold = a[i];
        for(j = i - 1; j >= 0 && a[j] > hold; j--) //change the sign of a[j] > hold;j--
        {
            a[j + 1] = a[j];
        }
        a[j + 1] = hold;
    }
}
